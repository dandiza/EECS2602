% 3B

% 2c has a greater magnitude cutoff than 2b, but converges at high
% frequecies. In addition,2b phase cut off in lower thatn 2c.

% 3c

t = -4:0.0001:4;
w = pi;
y = 0;
for n = -25:2:25
    if n == 0
        continue;
    else
        h1 = 8 / (20 + 1i * n * w);
        h2 = 8 / (2 + 1i * n * w);
        dn = (1 / (1i * w * n)) * (1 - exp(1i * -w * n));
        y = y + dn * h1 * h2 * exp(1j * n * w * t);
    end
end
figure(6)
plot(t, y);
title('y(t)');
xlabel('Time');
ylabel('Amplitude');
grid;
